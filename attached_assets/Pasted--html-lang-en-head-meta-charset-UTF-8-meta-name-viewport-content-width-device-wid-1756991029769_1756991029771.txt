  <script>
    // Phone input setup
    const input = document.querySelector("#phone");
    const iti = window.intlTelInput(input, {
      initialCountry: "in",
      utilsScript: "https://cdn.jsdelivr.net/npm/intl-tel-input@19.5.6/build/js/utils.js"
    });

    // Only numbers and limit digits as per country
    input.addEventListener('input', function () {
      let numbers = this.value.replace(/\D/g, '');
      let countryData = iti.getSelectedCountryData();
      let maxLen = 15;
      if (window.intlTelInputUtils && countryData.iso2) {
        let exampleNumber = intlTelInputUtils.getExampleNumber(countryData.iso2, true, intlTelInputUtils.numberType.MOBILE);
        if (exampleNumber) {
          let digits = exampleNumber.replace(/^\+\d+\s?/, '').replace(/\D/g, '');
          maxLen = digits.length;
        }
      }
      numbers = numbers.slice(0, maxLen);
      this.value = numbers;
    });

    // Toggle between sign up and sign in forms
    document.getElementById('show-signin').onclick = function() {
      document.getElementById('signup-form').style.display = 'none';
      document.getElementById('signin-form').style.display = 'flex';
    };
    document.getElementById('show-signup').onclick = function() {
      document.getElementById('signin-form').style.display = 'none';
      document.getElementById('signup-form').style.display = 'flex';
    };
  </script>
